## Important Compatibility Note

The current production `river_worker.py` is written for YOLO detection-style ONNX output. This package is a YOLO pose/keypoint model. Importing the asset is safe, but directly activating it may require worker-side pose output decoding support. Therefore the recommended flow is:

1. Upload zip to server.
2. Import to `/opt/river-watch/models/floater-pose-20260710` and `/home/ai-river/river-watch/models/floater-pose-20260710`.
3. Run ONNX shape/provider check.
4. Activate only after confirming compatibility or updating worker decoding.

## Server Safe Import Command

Run on `192.168.2.167`:

```bash
set -euo pipefail
cd /home/ai-river

ZIP="/home/ai-river/20260710_训练结果.zip"
EXPECTED_SHA="25edd848cad00c803dafd186e447e3b6c67a2ce150425873c766156d1a848f1b"
IMPORT_ID="floater-pose-20260710"
APP_DIR="/home/ai-river/river-watch"
OPT_MODELS="/opt/river-watch/models"
APP_MODELS="$APP_DIR/models"
STAGE="/home/ai-river/model-import-$IMPORT_ID"
BACKUP="/home/ai-river/model-import-backup-$IMPORT_ID-$(date +%Y%m%d-%H%M%S)"

echo "== 1. verify zip =="
[ -f "$ZIP" ] || { echo "missing $ZIP" >&2; exit 1; }
ACTUAL_SHA="$(sha256sum "$ZIP" | awk '{print $1}')"
echo "$ACTUAL_SHA  $ZIP"
[ "$ACTUAL_SHA" = "$EXPECTED_SHA" ] || { echo "SHA mismatch" >&2; exit 1; }

echo "== 2. extract =="
rm -rf "$STAGE"
mkdir -p "$STAGE"
unzip -o "$ZIP" -d "$STAGE"
ls -lh "$STAGE"/best.onnx "$STAGE"/best.pt "$STAGE"/data.yaml "$STAGE"/metadata.json

echo "== 3. backup current model dirs and env =="
mkdir -p "$BACKUP"
sudo mkdir -p "$OPT_MODELS" "$APP_MODELS"
sudo tar -C /opt/river-watch -czf "$BACKUP/opt-models-before-$IMPORT_ID.tar.gz" models 2>/dev/null || true
tar -C "$APP_DIR" -czf "$BACKUP/app-models-before-$IMPORT_ID.tar.gz" models 2>/dev/null || true
sudo tar -C /etc -czf "$BACKUP/etc-river-watch-before-$IMPORT_ID.tar.gz" river-watch 2>/dev/null || true

echo "== 4. install model assets without activating =="
sudo mkdir -p "$OPT_MODELS/$IMPORT_ID" "$APP_MODELS/$IMPORT_ID"
sudo install -m 0644 "$STAGE/best.onnx" "$OPT_MODELS/$IMPORT_ID/best.onnx"
sudo install -m 0644 "$STAGE/best.pt" "$OPT_MODELS/$IMPORT_ID/best.pt"
sudo install -m 0644 "$STAGE/last.pt" "$OPT_MODELS/$IMPORT_ID/last.pt"
sudo install -m 0644 "$STAGE/data.yaml" "$OPT_MODELS/$IMPORT_ID/data.yaml"
sudo install -m 0644 "$STAGE/keypoints.txt" "$OPT_MODELS/$IMPORT_ID/keypoints.txt"
sudo install -m 0644 "$STAGE/metadata.json" "$OPT_MODELS/$IMPORT_ID/metadata.json"
sudo install -m 0644 "$STAGE/status.json" "$OPT_MODELS/$IMPORT_ID/status.json"
sudo install -m 0644 "$STAGE/README_IMPORT.md" "$OPT_MODELS/$IMPORT_ID/README_IMPORT.md"
sudo cp -a "$STAGE"/*.png "$STAGE"/*.jpg "$STAGE"/*.csv "$STAGE"/*.yaml "$STAGE"/*.json "$APP_MODELS/$IMPORT_ID/" 2>/dev/null || true
sudo cp -a "$STAGE"/*.onnx "$STAGE"/*.pt "$STAGE"/*.txt "$STAGE"/README_IMPORT.md "$APP_MODELS/$IMPORT_ID/" 2>/dev/null || true

echo "== 5. checksums =="
sha256sum "$ZIP" | tee "$BACKUP/source-zip.sha256"
sha256sum "$OPT_MODELS/$IMPORT_ID/best.onnx" "$OPT_MODELS/$IMPORT_ID/best.pt" "$OPT_MODELS/$IMPORT_ID/data.yaml" | tee "$BACKUP/imported-model.sha256"

echo "== 6. ONNX provider/shape probe =="
PY="/opt/river-watch/ai-pipeline/.venv/bin/python"
[ -x "$PY" ] || PY="python3"
MODEL="$OPT_MODELS/$IMPORT_ID/best.onnx" "$PY" - <<'PY'
import os
import numpy as np
model = os.environ['MODEL']
try:
    import onnxruntime as ort
except Exception as exc:
    raise SystemExit(f"onnxruntime import failed: {exc}")
providers = [p for p in ["MIGraphXExecutionProvider", "CPUExecutionProvider"] if p in ort.get_available_providers()]
if not providers:
    providers = ort.get_available_providers()
print("available_providers=", ort.get_available_providers())
sess = ort.InferenceSession(model, providers=providers)
print("active_providers=", sess.get_providers())
print("inputs=", [(i.name, i.shape, i.type) for i in sess.get_inputs()])
print("outputs=", [(o.name, o.shape, o.type) for o in sess.get_outputs()])
shape = sess.get_inputs()[0].shape
h = int(shape[2] if isinstance(shape[2], int) else 640)
w = int(shape[3] if isinstance(shape[3], int) else 640)
x = np.zeros((1, 3, h, w), dtype=np.float32)
y = sess.run(None, {sess.get_inputs()[0].name: x})
print("runtime_output_shapes=", [tuple(v.shape) for v in y])
PY

echo "== imported but not activated =="
echo "model_dir=$OPT_MODELS/$IMPORT_ID"
echo "backup=$BACKUP"
```

## Optional Activation Command

Only run this after the ONNX shape check and worker compatibility are accepted.

```bash
set -euo pipefail
IMPORT_ID="floater-pose-20260710"
MODEL_ONNX="/opt/river-watch/models/$IMPORT_ID/best.onnx"
MODEL_LABELS="floater"

[ -f "$MODEL_ONNX" ] || { echo "missing $MODEL_ONNX" >&2; exit 1; }

sudo mkdir -p /etc/river-watch/model-env-backup-$(date +%Y%m%d-%H%M%S)
sudo cp -a /etc/river-watch/ai-worker-*.env /etc/river-watch/model-env-backup-$(date +%Y%m%d-%H%M%S)/ 2>/dev/null || true

for f in /etc/river-watch/ai-worker-CH0{1..8}.env /etc/river-watch/ai-worker-CH{01..08}.env; do
  [ -f "$f" ] || continue
  sudo sed -i \
    -e '/^YOLO_ONNX=/d' \
    -e '/^YOLO_LABELS=/d' \
    -e '/^YOLO_IMGSZ=/d' \
    -e '/^YOLO_CONF=/d' \
    -e '/^MODEL_REQUIRED=/d' "$f"
  {
    echo "YOLO_ONNX=$MODEL_ONNX"
    echo "YOLO_LABELS=$MODEL_LABELS"
    echo "YOLO_IMGSZ=640"
    echo "YOLO_CONF=0.25"
    echo "MODEL_REQUIRED=1"
  } | sudo tee -a "$f" >/dev/null
done

sudo systemctl restart river-ai-group@river-a river-ai-group@river-b
sleep 10
systemctl is-active river-ai-group@river-a river-ai-group@river-b
sudo journalctl -u river-ai-group@river-a -u river-ai-group@river-b --since '10 minutes ago' --no-pager | grep -Ei 'ONNX|provider|MIGraphX|CPUExecutionProvider|error|异常|failed' || true
```

## Rollback Activation

If activation causes worker errors, restore env backup and restart:

```bash
set -e
LATEST_BACKUP="$(ls -td /etc/river-watch/model-env-backup-* | head -1)"
sudo cp -a "$LATEST_BACKUP"/*.env /etc/river-watch/
sudo systemctl restart river-ai-group@river-a river-ai-group@river-b
systemctl is-active river-ai-group@river-a river-ai-group@river-b
```
